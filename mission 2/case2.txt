

int prime(int x)
{
	int abx;
	if (x >= 0)
		abx = x;
	else
		abx = -x;


	if (abx == 1 || abx == 0)
	{
		cout << x << "은(는) 소수가 아닙니다.\n";
		return 0;
	}
	else
	{
		int count = 0;
		for (int i = 1; i <= abx/2; i++)
		{
			if (abx % i == 0)
				count++;
		}
		if (count == 1)
			cout << x << "은(는) 소수입니다.\n";
		else
			cout << x << "은(는) 소수가 아닙니다.\n";

		return 0;
	}

}
